气泡插件在使用时有几点注意：
1.在head里需要引入prototype.js;
2.prototip插件与jQuery冲突，因此使用上会有些差别，具体见下面的例子；

1.javascript代码 
        var showTips;//全局方法变量，如页面不含jquery，直接new Tip()就可以
        (function($){
           $(document).ready(function(){
              using(['prototip','/js/plugins/prototip/styles.js'],function(){
                showTips = function(obj){
                   new Tip(obj, obj.getAttribute('tips'), {stem:'topLeft',title:'提示',style: 'protoblue',border:1,radius:2});
                }
              });
           });
        })(jQuery);//因prototip与jQuery冲突，所以与jquery相关的方法全部封闭处理
