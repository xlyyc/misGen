对话框 dialog.js

特殊说明对话框默认在当前的docment窗体对象上弹出，如果是Iframe集成需要将dialogToTop开关打开，弹出自适应后$对象将指向新的document，这个时候想操纵原页面的dom对象需要指定搜索范围"dialog.initDocument"。

JS部分

easyloader.isIframe = (window != top); //判断当前页面是否被Iframe集成
easyloader.dialogCssName = "dialog_style02";//启用SNS风格，动态程序在高层页面全局设定
$(document).ready(function () {
    using("dialog");
});
function show_dialog01() {
    $.ligerDialog.alert("这是一个提示弹出框");
}
function show_dialog03() {
    $.ligerDialog.confirm("这是一个确定框弹出框");
}
function show_dialog04() {
    $.ligerDialog.error("这是一个失败框弹出框");
}
function show_dialog05() {
    $.ligerDialog.open({
        width: 280,
        height: 100,
        content: "对话框弹出演示"
    });
}
function show_dialog06() {
    $.ligerDialog.ajaxFormSubmit({
        width:400,
        height:300,
        formUrl:"form1.jsp"
    });
}
function show_dialog07() {
    $.ligerDialog.prompt("输入框内容：", function (yes, value)
    {
        if (yes) alert("太棒了~！");
    });
}
function show_dialog08() {
    $.ligerDialog.question("这是一个问题弹出框");
}
function show_dialog09() {
    $.ligerDialog.success("这是一个成功弹出框");
}
function show_dialog10() {
    $.ligerDialog.waitting("10 秒后自动关闭",10000,true);
}
function show_dialog11() {
    $.ligerDialog.warn("这是一个警告弹出框");
}
function show_dialog12() {
    $.ligerDialog.warning("这是一个警告对话框");
}
