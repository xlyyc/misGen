javascript代码

 var menu1,menu2;
 $(document).ready(function(){
     using(['menu','menubar','dialog'],function(){
        menu1 = {width: 120, items:[
            { text: "增加", click: onclick11 ,icon:"add"},
            { text: "修改", click: onclick11,disable:true ,icon:"update"},
            { line: true },
            { text: "查看", click: onclick11 ,icon:"view"},
            { text: "关闭", click: onclick112,icon:"close" }
        ]
        };
        menu2 = {width:120, items:[
            {text:"添加",click:onclick11},
            {text:"子菜单",children:[
                {text:"word",click:onclick11},
                {text:"excel",click:onclick11,icon:"add"}
            ]
            }
        ]};
        $("#menubar").ligerMenuBar({ items: [
            { text: "文件", menu: menu1 },
            { text: "导出", menu: menu2 }
        ]
        });
    });
});

function onclick11(item, i){
    $.ligerDialog.success(item.text);
}
function onclick112(){
    menu1.hide();
}
