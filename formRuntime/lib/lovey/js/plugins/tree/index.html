JS部分(默认)带复选框和Icon

    using("tree",function(){
        $("#tree01").ligerTree({
        });
    });
 })

JS部分不带选框

 $(document).ready(function(){
    using("tree",function(){
        $("#tree02").ligerTree({
            checkbox: false
        });
    });
 })

JS部分简易模式

 $(document).ready(function(){
    using("tree",function(){
        $("#tree03").ligerTree({
            checkbox: false,
            parentIcon: null,
            childIcon: null
        });
    });
 })
