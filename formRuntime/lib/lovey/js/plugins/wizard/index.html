JS部分,引用wizard组件，并指定ID

$(document).ready(function () {
    using("wizard", function () {
        var wizardElement = $('#wizard');
        wizardElement.loveyInitWizard({
            items: [
                {stepId: "step1", active: "true", title: "步骤向导-1",nextFunction:function(){
                    alert("点击确定，进入下一步。。。");
                }},
                {stepId: "step2", title: "步骤向导-2"}
            ]
        });
    })
});
